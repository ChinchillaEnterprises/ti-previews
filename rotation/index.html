SVG + CSS
<svg viewBox="0 0 200 200" className="ti-mark">
  <g className="ti-ring">
    <circle cx="100" cy="100" r="75" fill="none"
            stroke="#101B3A" stroke-width="20" stroke-linecap="round"
            stroke-dasharray="78.54 78.54"
            transform="rotate(-60 100 100)" />
    <circle cx="100" cy="25"  r="10" fill="#101B3A" />
    <circle cx="165" cy="138" r="10" fill="#101B3A" />
    <circle cx="35"  cy="138" r="10" fill="#101B3A" />
  </g>
  <text x="100" y="120"
        font-family="Arial Black, sans-serif"
        font-weight="900" font-size="64"
        fill="#101B3A"
        text-anchor="middle"
        letter-spacing="-3">TI</text>
</svg>

.ti-mark .ti-ring {
  transform-origin: 100px 100px;
  transform-box: view-box;
  animation: ringSpin 22s linear infinite;
  will-change: transform;
}

@keyframes ringSpin {
  from { transform: rotate(0deg); }
  to   { transform: rotate(360deg); }
}

@media (prefers-reduced-motion: reduce) {
  .ti-mark .ti-ring { animation: none !important; }
}
